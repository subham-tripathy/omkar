wget https://storage.googleapis.com/flutter_infra_release/releases/stable/linux/flutter_linux_latest-stable.tar.xz
tar xf flutter_linux_latest-stable.tar.xz
export PATH="$PATH:`pwd`/flutter/bin"


sudo apt update
sudo apt install -y unzip openjdk-17-jdk wget

java -version

export JAVA_HOME=/usr/lib/jvm/java-17-openjdk-amd64
export PATH=$JAVA_HOME/bin:$PATH

mkdir -p $HOME/Android/cmdline-tools
cd $HOME/Android/cmdline-tools

wget https://dl.google.com/android/repository/commandlinetools-linux-11076708_latest.zip
unzip commandlinetools-linux-*_latest.zip
mv cmdline-tools latest

export ANDROID_HOME=$HOME/Android
export PATH=$ANDROID_HOME/cmdline-tools/latest/bin:$ANDROID_HOME/platform-tools:$PATH

source ~/.bashrc

sdkmanager --licenses
sdkmanager "platform-tools" "platforms;android-34" "build-tools;34.0.0"

flutter config --android-sdk $HOME/Android

flutter doctor
